        function renderAlbumGrid() {
            const grid = document.getElementById('album-grid');
            if (!grid) return;

            let htmlContent = '';
            const items = getAlbumItems(currentAlbumId);
            
            let useMasonry = true; 
            // 修改：对杂志排版强制移除瀑布流，使用规则整齐网格
            if (currentAlbumId === 'marcovaldo' || currentAlbumId === 'magazine') useMasonry = false;

            if (useMasonry) {
                grid.className = 'columns-1 sm:columns-2 md:columns-3 gap-8 space-y-8';
            } else {
                if (currentAlbumId === 'magazine') {
                    // 修改：杂志用两列显示以产生左右对称感
                    grid.className = 'grid grid-cols-1 sm:grid-cols-2 gap-8 md:gap-12';
                } else {
                    grid.className = 'grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 gap-8';
                }
            }

            items.forEach((item, index) => {
                const titleText = item.title[currentLang] || "";
                let currentImg = item.img || '3.jpg'; 

                let tagHTML = '';
                if (currentAlbumId === 'marcovaldo') {
                    tagHTML = `<span class="absolute bottom-3 right-3 bg-stone-900/10 text-stone-800/65 font-mono text-[9px] px-1.5 py-0.5 rounded backdrop-blur-xs group-hover:opacity-0 transition-opacity">CH ${item.ch}</span>`;
                }

                let topTag = currentAlbumId.split('_')[0].toUpperCase();
                if (currentAlbumId === 'marcovaldo') topTag = i18n[currentLang][`season${item.season === '春' ? 'Spring' : item.season === '夏' ? 'Summer' : item.season === '秋' ? 'Autumn' : 'Winter'}`];
                if (currentAlbumId === 'zine' || currentAlbumId === 'zine_poetry') topTag = 'ZINE PROJECT';
                if (currentAlbumId === 'magazine') topTag = 'MAGAZINE';
                if (currentAlbumId === 'exp_record_obs') topTag = 'OBSERVATION';

                // 增加骨架屏类
                const wrapperClass = useMasonry 
                    ? 'group relative overflow-hidden rounded-lg border border-stone-200/50 cursor-pointer inline-block w-full break-inside-avoid shadow-sm hover:shadow-md transition-shadow skeleton-bg min-h-[15rem]'
                    : 'group relative overflow-hidden rounded-lg border border-stone-200/50 aspect-[3/4] cursor-pointer flex items-center justify-center skeleton-bg';

                // 图片透明，并通过 transition-all 实现淡入
                const imgClass = useMasonry
                    ? 'w-full h-auto block transition-all duration-700 group-hover:scale-105 opacity-0'
                    : 'w-full h-full object-cover transition-all duration-700 group-hover:scale-102 opacity-0';

                let overlayHTML = '';
                if (titleText !== "") {
                    overlayHTML = `
                        <div class="mogu-overlay absolute inset-0 opacity-0 group-hover:opacity-100 flex flex-col justify-between p-6 text-center transition-opacity duration-300">
                            <div class="text-[9px] tracking-widest text-stone-400 font-light uppercase">${topTag}</div>
                            <h3 class="font-serif-art text-xs md:text-sm tracking-wide text-stone-800 leading-relaxed font-light px-2 my-auto">${titleText}</h3>
                            <div class="text-[8px] tracking-widest text-stone-400 uppercase font-light">HU ANGELA YEHUA</div>
                        </div>
                    `;
                } else {
                    overlayHTML = `<div class="absolute inset-0 bg-stone-900/10 opacity-0 group-hover:opacity-100 transition-opacity duration-300"></div>`;
                }

                // 添加 onload 淡入和骨架屏移除逻辑
                htmlContent += `
                    <div class="${wrapperClass}" onclick="openAlbumLightbox(${index})">
                        <img src="${currentImg}" class="${imgClass}" alt="${titleText}" loading="lazy" decoding="async" onload="this.classList.remove('opacity-0'); this.parentElement.classList.remove('skeleton-bg', 'min-h-[15rem]');" onerror="this.onerror=null; this.src='data:image/gif;base64,R0lGODlhAQABAIAAAAAAAP///yH5BAEAAAAALAAAAAABAAEAAAIBRAA7';">
                        ${tagHTML}
                        ${overlayHTML}
                    </div>
                `;
            });

            grid.innerHTML = htmlContent;
        }

        function openAlbumLightbox(index) {
            const items = getAlbumItems(currentAlbumId);
            const selectedItem = items[index];

            if (selectedItem.isSubAlbum) {
                enterAlbum(selectedItem.subAlbumId, true);
                return;
            }

            if (selectedItem.isDirectLightbox) {
                lightboxActiveItems = [{
                    img: selectedItem.lightboxImg || selectedItem.img,
                    title: selectedItem.title[currentLang],
                    meta: '',
                    genre: currentAlbumId.toUpperCase()
                }];
                lightboxCurrentIndex = 0;
                openLightboxModal();
                return;
            }

            lightboxActiveItems = items.map((item, i) => {
                let genre = currentAlbumId.split('_')[0].toUpperCase();
                let meta = item.title[currentLang];
                
                if (currentAlbumId === 'marcovaldo') {
                    genre = i18n[currentLang][`season${item.season === '春' ? 'Spring' : item.season === '夏' ? 'Summer' : item.season === '秋' ? 'Autumn' : 'Winter'}`];
                    meta = `Marcovaldo · CH ${item.ch}`;
                }
                if (currentAlbumId === 'exp_record_obs') {
                    genre = "NOTE";
                    meta = `Record No.${i+1}`;
                }

                return { img: item.img || '3.jpg', title: item.title[currentLang], meta: meta, genre: genre };
            });

            lightboxCurrentIndex = index;
            openLightboxModal();
        }

        function openLightboxModal() {
            const modal = document.getElementById('lightbox-modal');
            modal.classList.remove('hidden'); modal.classList.add('flex');
            document.addEventListener('keydown', handleLightboxKeyboard);
            updateLightboxContent();
        }

        function updateLightboxContent() {
            if (lightboxActiveItems.length === 0) return;
            const currentItem = lightboxActiveItems[lightboxCurrentIndex];

            const img = document.getElementById('lightbox-img');
            const title = document.getElementById('lightbox-title');
            const meta = document.getElementById('lightbox-meta');
            const genre = document.getElementById('lightbox-genre');

            // 恢复灯箱大图的骨架屏状态
            img.classList.add('opacity-0');
            const imgWrapper = document.getElementById('lightbox-img-wrapper');
            imgWrapper.classList.add('skeleton-bg-dark');

            img.onload = function() {
                this.classList.remove('opacity-0');
                imgWrapper.classList.remove('skeleton-bg-dark');
            };

            img.src = currentItem.img;
            title.innerText = currentItem.title;
            meta.innerText = currentItem.meta;
            genre.innerText = currentItem.genre.toUpperCase();

            const prevBtn = document.getElementById('lightbox-prev-btn');
            const nextBtn = document.getElementById('lightbox-next-btn');
            if (lightboxActiveItems.length <= 1) {
                prevBtn.classList.add('hidden'); nextBtn.classList.add('hidden');
            } else {
                prevBtn.classList.remove('hidden'); nextBtn.classList.remove('hidden');
            }

            resetZoom();
        }

        function navigateLightbox(direction) {
            if (lightboxActiveItems.length === 0) return;
            lightboxCurrentIndex += direction;
            if (lightboxCurrentIndex >= lightboxActiveItems.length) lightboxCurrentIndex = 0; 
            else if (lightboxCurrentIndex < 0) lightboxCurrentIndex = lightboxActiveItems.length - 1; 
            updateLightboxContent();
        }

        function handleLightboxKeyboard(e) {
            const modal = document.getElementById('lightbox-modal');
            if (modal.classList.contains('hidden')) return;
            if (e.key === 'ArrowRight' || e.key === 'd' || e.key === 'D') navigateLightbox(1);
            else if (e.key === 'ArrowLeft' || e.key === 'a' || e.key === 'A') navigateLightbox(-1);
            else if (e.key === 'Escape') closeLightbox();
        }

        function closeLightbox() {
            const modal = document.getElementById('lightbox-modal');
            modal.classList.add('hidden'); modal.classList.remove('flex');
            document.removeEventListener('keydown', handleLightboxKeyboard);
        }

        function zoomImage(e) {
            const container = e.currentTarget;
            const img = container.querySelector('.zoom-img');
            const rect = container.getBoundingClientRect();
            const x = ((e.clientX - rect.left) / rect.width) * 100;
            const y = ((e.clientY - rect.top) / rect.height) * 100;
            img.style.transformOrigin = `${x}% ${y}%`;
            img.style.transform = 'scale(2.2)';
        }

        function resetZoom() {
            const img = document.getElementById('lightbox-img');
            if (img) { img.style.transform = 'scale(1)'; img.style.transformOrigin = 'center center'; }
        }

        function initScrollIndicator() {
            const indicator = document.getElementById('scroll-indicator');
            const scrollText = document.getElementById('scroll-text');
            const linesBox = document.getElementById('scroll-lines-box');
            if (!indicator || !scrollText || !linesBox) return;

            indicator.addEventListener('click', () => {
                if (window.scrollY > 300) window.scrollTo({ top: 0, behavior: 'smooth' });
                else { const section = document.getElementById('works'); if (section) section.scrollIntoView({ behavior: 'smooth' }); }
            });

            window.addEventListener('scroll', () => {
                if (window.scrollY > 300) {
                    scrollText.innerText = i18n[currentLang].scrollUp;
                    linesBox.style.transform = 'rotate(180deg)'; 
                } else {
                    scrollText.innerText = i18n[currentLang].scroll;
                    linesBox.style.transform = 'rotate(0deg)'; 
                }
            });
        }
